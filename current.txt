class LibrarayManager{
    books;

    constructor(){
        this.books = [];
    }
    addBook(title,author){
        const book = this.books.find(b=>b.title === title);
        if(book){
            console.log( `Book already exists: ${title}`);
            return { status: 'error'};
        }
        this.books.push({title,author});
        console.log(`Book added: ${title} ${author}`);
        return {status:'success'};
    }
    listBooks(){
        console.log('Listing all books');
        return this.books;
    }
    removeBook(title){
        const index = this.books.findIndex(b=>b.title === title);
        if(index === -1) {
            console.log(`Book not found: ${title}`);
            return { status:'error'};
        }
        this.books.splice(index,1);
        console.log(`Book removed: ${title}`);
        return { status:'success'};
    }
}
module.exports = LibrarayManager;
