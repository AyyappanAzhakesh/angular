class Inventory {
    constructor() {
        this.products = new Map();
    }

    addProduct(name, quantity) {
        const currentQuantity = this.products.get(name) || 0;
        this.products.set(name, currentQuantity + quantity);
    }

    updateProduct(name, quantity) {
        if (!this.products.has(name)) {
            throw new Error('Product not found');
        }
        this.products.set(name, quantity);
    }

    removeProduct(name) {
        if (!this.products.has(name)) {
            throw new Error('Product not found');
        }
        this.products.delete(name);
    }

    getProduct(name) {
        return this.products.has(name) ? this.products.get(name) : null;
    }
}

module.exports = Inventory;
