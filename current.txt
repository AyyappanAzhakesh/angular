class ShoppingList {
  constructor() {
    this.items = [];
  }

  addItem(name, quantity) {
    this.items.push({ name, quantity });
  }

  updateItem(name, quantity) {
    const item = this.items.find(i => i.name === name);
    if (item) {
      item.quantity = quantity;
    }
    // silently ignore if item not found (as expected by tests)
  }

  deleteItem(name) {
    this.items = this.items.filter(i => i.name !== name);
    // safely handles missing item (as expected)
  }

  getList() {
    return this.items;
  }
}

module.exports = new ShoppingList();
