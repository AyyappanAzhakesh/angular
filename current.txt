class Destination {
  constructor(name, cost) {
    this.name = name;
    this.cost = cost;
  }
}

class TravelPlan {
  constructor() {
    this.destinations = [];
  }

  addDestination(destination) {
    this.destinations.push(destination);
  }

  calculateTotalCost() {
    return this.destinations.reduce((total, destination) => total + destination.cost, 0);
  }

  printPlan() {
    console.log("Travel Plan:");
    this.destinations.forEach(destination => {
      console.log(`${destination.name} - $${destination.cost.toFixed(2)}`);
    });
  }
}

// Attach classes to global for compatibility with the test file
global.Destination = Destination;
global.TravelPlan = TravelPlan;

module.exports = {
  Destination,
  TravelPlan
};
