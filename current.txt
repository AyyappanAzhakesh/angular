class Product {
    constructor(name, price) {
        this.name = name;
        this.price = price;
    }
}

class ShoppingCart {
    constructor() {
        this.items = [];
    }

    addItem(product) {
        this.items.push(product);
    }

    calculateTotalPrice() {
        return this.items.reduce((total, item) => total + item.price, 0);
    }
}

// Make them available in global scope for testing
global.Product = Product;
global.ShoppingCart = ShoppingCart;

module.exports = { Product, ShoppingCart };
