class QuoteManager {
  constructor() {
    this.quotes = [];
    this.nextId = 1;
  }

  addQuote(text, author) {
    const quote = {
      id: this.nextId++,
      text,
      author
    };
    this.quotes.push(quote);
    return quote;
  }

  updateQuote(id, newText, newAuthor) {
    const quote = this.quotes.find(q => q.id === id);
    if (!quote) return null;
    quote.text = newText;
    quote.author = newAuthor;
    return quote;
  }

  deleteQuote(id) {
    const index = this.quotes.findIndex(q => q.id === id);
    if (index === -1) return false;
    this.quotes.splice(index, 1);
    return true;
  }

  listQuotes() {
    return this.quotes;
  }
}

module.exports = QuoteManager;
