class Destination {
  constructor(name, cost) {
    this.name = name;
    this.cost = cost;
  }
}

class TravelPlan {
  constructor() {
    this.destinations = [];
  }

  addDestination(destination) {
    this.destinations.push(destination);
  }

  calculateTotalCost() {
    return this.destinations.reduce((total, dest) => total + dest.cost, 0);
  }

  printPlan() {
    console.log("Travel Plan:");
    this.destinations.forEach(dest => {
      console.log(`${dest.name} - $${dest.cost.toFixed(2)}`);
    });
  }
}

module.exports = {
  Destination,
  TravelPlan
};
