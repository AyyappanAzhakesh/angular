class ShoppingList {
    constructor() {
        this.groceries = [];
        this.totalPrice = 0;
    }
    addItem(itemName, price) {
        this.groceries.push({ itemName, price });
    }
    calculateTotalPrice() {
        let result = this.groceries.reduce((item, price) => item + price.price, 0);
        this.totalPrice = result;
    }
    removeItem(itemName) {
        this.groceries = this.groceries.filter(p => p.itemName !== itemName);
    }
    clearList() {
        this.groceries = [];
        this.totalPrice = 0;
    }
    getItems() {
        return this.groceries;
    }
    getTotalPrice() {
        return this.totalPrice;
    }
}
module.exports = ShoppingList;
