class ContactManager {
  constructor() {
    this.contacts = [];
  }

  addContact(name, phone) {
    const exists = this.contacts.some(contact => contact.name === name);
    if (exists) {
      return { status: 'error', message: 'Contact already exists' };
    }
    this.contacts.push({ name, phone });
    return { status: 'success', message: 'Contact added successfully' };
  }

  removeContact(name) {
    const index = this.contacts.findIndex(contact => contact.name === name);
    if (index === -1) {
      return { status: 'error', message: 'Contact not found' };
    }
    this.contacts.splice(index, 1);
    return { status: 'success', message: 'Contact removed successfully' };
  }

  listContacts() {
    return this.contacts;
  }
}

module.exports = ContactManager;
