import { Component } from '@angular/core';
import { NgForm } from '@angular/forms'; // ðŸ‘ˆ Allows Angular to understand form info

@Component({
  selector: 'app-eventform',
  templateUrl: './eventform.component.html',
  styleUrls: ['./eventform.component.css']
})
export class EventformComponent {

  // ðŸŒŸ This method is triggered when the form is submitted
  onSubmit(form: NgForm) {
    const data = form.value;

    // ðŸ“¢ Show the submitted information
    alert(`Registration Details:
    Name: ${data.name}
    Email: ${data.email}
    Phone: ${data.phone}
    Event: ${data.event}
    Dietary Preferences: ${data.diet}`);

    // ðŸ”„ Reset the form after submission
    form.reset();
  }
}
